# FunÃ§Ã£o: obtÃ©m tamanho do arquivo de forma portÃ¡til (macOS/Linux)
get_size_bytes() {
    local f="$1"
    if size=$(stat -f%z "$f" 2>/dev/null); then
        echo "$size"
    else
        stat -c%s "$f" 2>/dev/null || echo "0"
    fi
}

# FunÃ§Ã£o: formata bytes para exibiÃ§Ã£o legÃ­vel
format_bytes() {
    local bytes=$1
    if [ $bytes -lt 1024 ]; then
        echo "${bytes}B"
    elif [ $bytes -lt 1048576 ]; then
        echo "$((bytes / 1024))KB"
    else
        echo "$((bytes / 1048576))MB"
    fi
}

# FunÃ§Ã£o: verifica se um arquivo deve ser ignorado
should_ignore_file() {
    local filepath="$1"
    local filename=$(basename "$filepath")
    
    # Sempre ignorar .DS_Store
    if [[ "$filename" == ".DS_Store" ]]; then
        return 0
    fi
    
    # Converte pattern em array
    IFS='|' read -ra IGNORE_FILES_ARRAY <<< "$IGNORE_FILES_PATTERN"
    
    for pattern in "${IGNORE_FILES_ARRAY[@]}"; do
        # Remove espaÃ§os em branco
        pattern=$(echo "$pattern" | xargs)
        
        # Verifica se o pattern corresponde ao nome do arquivo
        if [[ "$filename" == $pattern ]]; then
            return 0  # Deve ignorar
        fi
    done
    
    return 1  # NÃ£o deve ignorar
}

# FunÃ§Ã£o: verifica se um caminho deve ser ignorado
should_ignore_path() {
    local filepath="$1"
    local project_dir="$2"
    
    # ObtÃ©m o caminho absoluto do arquivo
    local absolute_filepath="$(realpath "$filepath" 2>/dev/null || echo "$filepath")"
    
    # Remove o diretÃ³rio do projeto do caminho para comparaÃ§Ã£o relativa
    local relative_path="${filepath#$project_dir/}"
    
    # Verifica caminhos ABSOLUTOS especÃ­ficos
    if [ -n "$IGNORE_ABSOLUTE_PATHS" ]; then
        IFS='|' read -ra IGNORE_ABS_ARRAY <<< "$IGNORE_ABSOLUTE_PATHS"
        for ignore_path in "${IGNORE_ABS_ARRAY[@]}"; do
            ignore_path=$(echo "$ignore_path" | xargs)
            # Verifica se o caminho absoluto comeÃ§a com o caminho a ignorar
            if [[ "$absolute_filepath" == "$ignore_path"* ]]; then
                return 0  # Deve ignorar
            fi
        done
    fi
    
    # Verifica caminhos RELATIVOS especÃ­ficos
    if [ -n "$IGNORE_PATHS" ]; then
        IFS='|' read -ra IGNORE_PATHS_ARRAY <<< "$IGNORE_PATHS"
        for ignore_path in "${IGNORE_PATHS_ARRAY[@]}"; do
            ignore_path=$(echo "$ignore_path" | xargs)
            # Verifica se o caminho relativo contÃ©m o padrÃ£o
            if [[ "$relative_path" == *"$ignore_path"* ]]; then
                return 0  # Deve ignorar
            fi
        done
    fi
    
    return 1  # NÃ£o deve ignorar
}

                \( -type f -name '.DS_Store' -prune \) -o \

    } >> "$output_file"

        -type f \( "${code_name_expr[@]}" \) -print0 2>/dev/null \
        | sort -z \

            RELATIVE_PATH="./${filepath#$project_dir/}"

                ((skipped_count++))
                continue
            fi
            
            # ObtÃ©m tamanho do arquivo
            SIZE_BYTES="$(get_size_bytes "$filepath")"
            SIZE_FORMATTED="$(format_bytes $SIZE_BYTES)"
            
            # Verifica limite de tamanho

                {

                    echo "â””â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€"
                    echo
                } >> "$output_file"
                ((skipped_count++))
                continue
            fi
            
            # Adiciona conteÃºdo do arquivo
            {
                echo "â”Œâ”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€"

                    tr -d '\r' < "$filepath" 2>/dev/null | nl -ba -w4 -s' â”‚ ' || echo "â”‚ [Erro ao ler arquivo]"

            ((file_count++))

        done
    
    # Adiciona sumÃ¡rio ao final do arquivo
    {

        # echo "ðŸ“Š SUMÃRIO"
